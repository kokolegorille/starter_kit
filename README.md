# Electron starter kit

Starter Kit for Electron

## Npm
$ mkdir starter_kit
$ cd starter_kit
$ npm init -y

$ vim package.json

$ npm install

## Git
$ vim .gitignore
npm-debug.log
/node_modules
/build/*.js
/build/*.css
/src/js/config/*

$ git init
$ git add .
$ git commit -m "Initial commit"

## Webpack
$ vim webpack.config.js

$ mkdir -p src/js/config
$ mkdir src/css
$ mkdir build

$ vim build/index.html
